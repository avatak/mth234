\question Let \(\br(u,v)=\gen{u+v,u-v,u^2-v^2}\)
\begin{parts}
    \part Evaulate \(\br(2,-1)\) and \(\br(-1,2)\).
        \ifprintanswers
            \begin{solution}
                \begin{align*}
                    \br(2,-1) & = (2-1,2-(-1),2^2-(-1)^2)\\
                        & = (1,3,3)\\
                    \br(-1,2) & = (-1+2,-1-2,(-1)^2-2^2)\\
                        & = (1,-3,-3)
                \end{align*}
            \end{solution}
        \else
            \vfill
        \fi
    \part Find \(u,v\) so that \(\br(u,v)=(3,-1,-3)\).
        \ifprintanswers
            \begin{solution}
                If \(u+v=3\) then \(v=3-u\). Then
                \[
                    u-v=-1 \bif u-(3-u)=-1 \bif 2u=2\bif u=1
                \]
                Which implies \(v=3-1=2\).
                To verify \(u=1,v=2\) is correct,
                \[
                    \br(1,2)=(1+2,1-2,(1)^2-(2)^2)=(3,-1,-3).
                \]
            \end{solution}
        \else
            \vfill
        \fi
        


    \part Show that \((0,0,1)\) is not a point on the surface.
        \ifprintanswers
            \begin{solution}
                Since \(u-v=0\), \(u=v\). Then \(u+v=0\) gives \(u+u=0\) which means \(u=0=v\).
                But \(\br(0,0)=(0,0,0)\ne (0,0,1)\).
            \end{solution}
        \else
            \vfill
        \fi

    % \(\br(u,v)=\gen{u+v,u-v,u^2-v^2}\)
    \part Recall that a surface \(\br(u,v)\) is \emph{smooth} if \(\br_u\) and \(\br_v\) are both continuous and 
    \(|\br_u\times\br_v|\) is never \(0\) for \((u,v)\) in the interior of the domain 
    (this means that at any point on the surface, the normal vector to the tangent plane is not \(\bm{0}\)).
    Show that \(\br(u,v)\) is continuous.
        \ifprintanswers
            \begin{solution}
            We first compute \(\br_u\), \(\br_v\), \(\br_u\times\br_v\), and \(|\br_u\times\br_v|\).
            \begin{align*}
            \br_u(u,v) & = \gen{
                                1+0,1-0,2u-0
                               }\\
                        & = \gen{1,1,2u}\\
            \br_v(u,v)  & = \gen{0+1,0-1,0-2v}\\
                        & = \gen{1,-1,-2v}.\\
            \br_u\times\br_v & = 
            \left|\begin{NiceMatrix}
                \bi & \bj & \bk \\
                1 & 1 & 2u \\
                1 & -1 & -2v\\
            \end{NiceMatrix}\right|\\
            & = \gen{2u-2v,2u+2v,-2}\\
            |\br_u\times\br_v| & = \sqrt{(2u-2v)^2+(2u+2v)^2+(-2)^2}\\
                & = \sqrt{
                    4u^2-4uv+4v^2+4u^2+4uv+4v^2+4
                }\\
                & = \sqrt{8u^2+8v^2+4}
            \end{align*}
            It is clear that \(\br_u\)
            and \(\br_v\) are continuous everywhere (constant functions and polynomials). 
            The only way \(|\br_u\times\br_v|=\sqrt{8u^2+8v^2+4}=0\) is if \(8u^2+8v^2+4=0\). Since \(8u^2+8v^2+4\ge4\) for any \((u,v)\) we know \(|\br_u\times\br_v|\) is never 0.

            This means the surface is continuous

            \end{solution}
        \else
            \vfill
        \fi

    \part Find an equation of the tangent plane to the parametric surface \(\br(u,v)\) at the point 
    \(u=1\) and \(v=-1\).
        \ifprintanswers
            \begin{solution}
                A point on the tangent plane is given by 
                \(\br(1,-1)=(0,2,0)\). Using \(\br_u\times\br_v\) found above,
                \begin{align*}
                    \br_u\times\br_v(1,-1) & = \gen{2(1)-2(-1),2(1)+2(-1),-2}\\
                    & = \gen{4,0,-2}
                \end{align*}
                Therefore an equation for the tangent plane is
                \begin{align*}
                    \gen{4,0,-2}\cdot\gen{x-0,y-2,z-0} & = 0\\
                    4(x-0)+0(y-2)-2(z-0) & = 0 \\
                    4x-2z & = 0
                \end{align*}
            \end{solution}
        \else
            \vfill
            \newpage
        \fi


    \end{parts}

\question Find two parametric representations for the part of the plane \(z=3x+5y\) that lies inside the cylinder \(x^2+y^2=4\) by completing the following
\begin{parts}
    \part 
    \[ 
        \br_1(u,v) = \gen{ u ,\underline{~~~~~~~~~~},\underline{~~~~~~~~~~}}\qquad \text{with}\quad u^2+v^2\le \underline{~~~~~~~~~~}
    \]
    \ifprintanswers
        \begin{solution}
            \[
                \br_1(u,v) = \gen{u,v,3u+5v},\quad u^2+v^2\le 4
            \]
        \end{solution}
    \else
        \vfill
    \fi
    \part 
    \[
        \br_2(s,t) = \gen{s\cos(t),s \sin(t), \underline{~~~~~~~~~~}}\quad \text{with}\quad s\in [\underline{~~~~~},\underline{~~~~~}],\quad t\in [\underline{~~~~~},\underline{~~~~~}]
    \]
    \ifprintanswers
        \begin{solution}
            \[
                \br_2(s,t) = \gen{s\cos(t),s\sin(t),3s\cos(t)+5s\sin(t)},\quad s\in[0,2],\quad t\in [0,2\pi]
                \]
        \end{solution}
    \else
        \vfill
    \fi
\end{parts}

\question Let \(S\) be the surface consisting of the part of the sphere \(x^2+y^2+z^2=16\) that lies between the planes 
\(z=-2\) and \(z=2\).

\begin{parts}
    \part Find a parametric representation \(\br(u,v)\) of \(S\) by completing the following
        \[
            \br(u,v) = \gen{4\cos(u)\sin(v),\underline{~~~~~},\underline{~~~~~}}
        \]
            \[ u\in[0,2\pi]\quad \text{and}\quad v\in[\underline{~~~~~},\underline{~~~~~}]\]
        \ifprintanswers
        \begin{solution}
        \[
            \br(u,v) = \gen{4\cos(u)\sin(v),4\sin(u)\sin(v),4\cos(v)}
        \]
        \[ u\in[0,2\pi]\quad \text{and}\quad v\in[\pi/3,2\pi/3]\]
        \end{solution}
    \else
        \vfill
    \fi
    \part Find a parametric representation of the boundary curve of \(S\) where \(z=2\)
        \[
            \br_1(t)=\gen{\underline{~~~~~},\underline{~~~~~},2},\quad t\in[0,2\pi]
        \]
    \ifprintanswers
        \begin{solution}
        If \(z=2\), our equation becomes \(x^2+y^2=12\), i.e. a circle with radius \(2\sqrt{3}\).
        \[
              \br_1(t)=\gen{2\sqrt{3}\cos(t),2\sqrt{3}\sin(t),2},\quad t\in[0,2\pi]
        \]
        \end{solution}
    \else
        \vfill
    \fi
    \part Find a parametric representation of the boundary curve of \(S\) where \(z=-2\)
    \[
        \br_2(t)=\gen{\underline{~~~~~},\underline{~~~~~},-2},\quad t\in[0,2\pi]
    \]
        \ifprintanswers
        \begin{solution}
            \[
                \br_2(t)=\gen{2\sqrt{3}\cos(t),2\sqrt{3}\sin(t),-2}, \quad t\in[0,2\pi]
            \]
        \end{solution}
    \else
        \vfill
    \fi
\end{parts}
